// ############### IR REMOTE CONTROL Code File IR - NEC Standard ############### //
// First byte ADRESS, Second its a command (B914 - address B9, cmd 14)           //
// Default is Kenwood RC-406, small portable remote from car audio systems	 //
rcCmdVolumeUp   0xB914;  // Glosnosc + / Volume +
rcCmdVolumeDown 0xB915;  // Glosnosc - / Volume -
rcCmdArrowRight 0xB90B;  // RIGHT arrow -  next station / strzałka w prawo - nastepna stacja
rcCmdArrowLeft  0xB90A;  // LEFT arrow - previus station / strzałka w lewo - poprzednia stacja  
rcCmdArrowUp    0xB987;  // UP arrow - station list up / strzałka w góre - lista stacji krok do gory
rcCmdArrowDown  0xB986;  // DOWN arrow  - station list down / strzałka w dół - lista stacj krok na dół
rcCmdBack       0xB985;  // Back button / Przycisk powrotu
rcCmdOk         0xB90E;  // EnterOK button  - confirmation / Przycisk Ent - zatwierdzenie stacji
rcCmdSrc        0xB913;  // Display modes / Tryby wyswietlacza
rcCmdMute       0xB916;  // Mute / Wyciszenie dzwieku
rcCmdAud        0xB917;  // Equalizer/ Equalizer dzwieku
rcCmdDirect     0xB90F;  // Display Dimmer, Network bank update / Janość ekranu, dwa tryby 1/16 lub pełna janość    
rcCmdBankMinus  0xB90C;  // Station Bank Memory select / Wysweitla wybór banku
rcCmdBankPlus   0xB90D;  // Station Bank Memory select / Wysweitla wybór banku
rcCmdRed        0xB988;  // Power ON/OFF  / Tryb Standby Wlacz/Wylacz
rcCmdGreen      0xB992;  // Sleep / Funkcja drzemki
rcCmdKey0       0xB900;  // Button / Przycisk "0"
rcCmdKey1       0xB901;  // Button / Przycisk "1"
rcCmdKey2       0xB902;  // Button / Przycisk "2"
rcCmdKey3       0xB903;  // Button / Przycisk "3"
rcCmdKey4       0xB904;  // Button / Przycisk "4"
rcCmdKey5       0xB905;  // Button / Przycisk "5"
rcCmdKey6       0xB906;  // Button / Przycisk "6"
rcCmdKey7       0xB907;  // Button / Przycisk "7"
rcCmdKey8       0xB908;  // Button / Przycisk "8"
rcCmdKey9       0xB909;  // Button / Przycisk "9"
